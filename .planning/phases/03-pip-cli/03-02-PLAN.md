    Key implementation details:
    - SENTINEL_START = "<!-- goodvibes:start -->"
    - SENTINEL_END = "<!-- goodvibes:end -->"
    - extract_version(block: str) -> str | None: re.search(r"# goodvibes: v([\d.]+)", block)
    - version_gte(a: str, b: str) -> bool: split on ".", convert to int list with zero-padding
      to equal length, compare element by element — handles "1.10.0 > 1.9.0" correctly
    - _extract_sentinel_block(content: str) -> str: find SENTINEL_START and SENTINEL_END,
      return content[start_idx:end_idx + len(SENTINEL_END)]; return "" if either not found
    - merge_claude(dest_path: pathlib.Path, template_content: str) -> None:
      Case A: not dest_path.exists() → dest_path.parent.mkdir(parents=True, exist_ok=True);
              dest_path.write_text(template_content, encoding="utf-8"); return
      Case B: dest exists, SENTINEL_START not in existing →
              write existing.rstrip() + "\n\n" + template_block + "\n"
      Case C: dest exists, sentinel present, existing_ver < template_ver →
              before = existing[:start_idx]; after = existing[end_idx + len(SENTINEL_END):]
              write before + template_block + after
      Case D: dest exists, sentinel present, existing_ver >= template_ver → return (no write)

    Security: Case B idempotency is guaranteed by Case D — the second call will find the
    sentinel block written by the first call and skip (version equal). No extra duplicate check needed.

    Sentinel format is byte-for-byte identical to TypeScript output — the test suite uses
    the same TEMPLATE_CONTENT constant as Phase 2 (sentinel-merge.test.ts).

        def ignore_fn(directory: str, contents: list[str]) -> set[str]:
          ignored: set[str] = set()
          for name in contents:
            full = pathlib.Path(directory) / name
            try:
              rel = full.relative_to(template_dir)
            except ValueError:
              ignored.add(name)
              continue
            if name == "CLAUDE.md":
              ignored.add(name)   # sentinel merge handles it separately

              ignored.add(name)   # path traversal guard

              ignored.add(name)
            # No-clobber: skip if destination already exists
            dest_candidate = dest_dir / rel
            if dest_candidate.exists():
              ignored.add(name)
          return ignored
